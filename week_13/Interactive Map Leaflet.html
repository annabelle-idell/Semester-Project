US_cities <- us_cities() %>%
  filter(!state_name %in% c("Puerto Rico", "Alaska", "Hawaii", "District of Columbia")) %>% 
  select(c(city, state_name, population)) %>% 
  group_by(state_name) %>% 
  slice_max(population, n = 3) %>% 
  arrange(desc(population), .by_group = TRUE) %>% 
  mutate(rank = c(1, 2, 3))
